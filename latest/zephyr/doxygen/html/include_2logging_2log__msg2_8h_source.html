105 enum z_log_msg2_mode {

107  * treated as a fallback method when others cannot be used.
108  */
109  Z_LOG_MSG2_MODE_RUNTIME,

138 #ifdef CONFIG_LOG2_USE_VLA
139 #define Z_LOG_MSG2_ON_STACK_ALLOC(ptr, len) \
140  long long _ll_buf[ceiling_fraction(len, sizeof(long long))]; \
141  long double _ld_buf[ceiling_fraction(len, sizeof(long double))]; \
142  ptr = (sizeof(long double) == Z_LOG_MSG2_ALIGNMENT) ? \
143  (struct log_msg2 *)_ld_buf : (struct log_msg2 *)_ll_buf; \
144  if (IS_ENABLED(CONFIG_LOG_TEST_CLEAR_MESSAGE_SPACE)) { \
145  /* During test fill with 0's to simplify message comparison */ \
146  memset(ptr, 0, len); \
147  }
148 #else /* Z_LOG_MSG2_USE_VLA */
149 /* When VLA cannot be used we need to trick compiler a bit and create multiple
150  * fixed size arrays and take the smallest one that will fit the message.
151  * Compiler will remove unused arrays and stack usage will be kept similar
152  * to vla case, rounded to the size of the used buffer.
153  */
154 #define Z_LOG_MSG2_ON_STACK_ALLOC(ptr, len) \
155  long long _ll_buf32[32 / sizeof(long long)]; \
156  long long _ll_buf48[48 / sizeof(long long)]; \
157  long long _ll_buf64[64 / sizeof(long long)]; \
158  long long _ll_buf128[128 / sizeof(long long)]; \
159  long long _ll_buf256[256 / sizeof(long long)]; \
160  long double _ld_buf32[32 / sizeof(long double)]; \
161  long double _ld_buf48[48 / sizeof(long double)]; \
162  long double _ld_buf64[64 / sizeof(long double)]; \
163  long double _ld_buf128[128 / sizeof(long double)]; \
164  long double _ld_buf256[256 / sizeof(long double)]; \
165  if (sizeof(long double) == Z_LOG_MSG2_ALIGNMENT) { \
166  ptr = (len > 128) ? (struct log_msg2 *)_ld_buf256 : \
167  ((len > 64) ? (struct log_msg2 *)_ld_buf128 : \
168  ((len > 48) ? (struct log_msg2 *)_ld_buf64 : \
169  ((len > 32) ? (struct log_msg2 *)_ld_buf48 : \
170  (struct log_msg2 *)_ld_buf32)));\
171  } else { \
172  ptr = (len > 128) ? (struct log_msg2 *)_ll_buf256 : \
173  ((len > 64) ? (struct log_msg2 *)_ll_buf128 : \
174  ((len > 48) ? (struct log_msg2 *)_ll_buf64 : \
175  ((len > 32) ? (struct log_msg2 *)_ll_buf48 : \
176  (struct log_msg2 *)_ll_buf32)));\
177  } \
178  if (IS_ENABLED(CONFIG_LOG_TEST_CLEAR_MESSAGE_SPACE)) { \
179  /* During test fill with 0's to simplify message comparison */ \
180  memset(ptr, 0, len); \
181  }
182 #endif /* Z_LOG_MSG2_USE_VLA */

195 #define Z_LOG_MSG2_STACK_CREATE(_domain_id, _source, _level, _data, _dlen, ...)\
196 do { \
197  int _plen; \
198  if (GET_ARG_N(1, __VA_ARGS__) == NULL) { \
199  _plen = 0; \
200  } else { \
201  CBPRINTF_STATIC_PACKAGE(NULL, 0, _plen, Z_LOG_MSG2_ALIGN_OFFSET, \
202  0, __VA_ARGS__); \
203  } \
204  struct log_msg2 *_msg; \
205  Z_LOG_MSG2_ON_STACK_ALLOC(_msg, Z_LOG_MSG2_LEN(_plen, 0)); \
206  if (_plen) { \
207  CBPRINTF_STATIC_PACKAGE(_msg->data, _plen, \
208  _plen, Z_LOG_MSG2_ALIGN_OFFSET, \
209  0, __VA_ARGS__);\
210  } \
211  struct log_msg2_desc _desc = \
212  Z_LOG_MSG_DESC_INITIALIZER(_domain_id, _level, \
213  (uint32_t)_plen, _dlen); \
214  LOG_MSG2_DBG("creating message on stack: package len: %d, data len: %d\n", \
215  _plen, (int)(_dlen)); \
216  z_log_msg2_static_create((void *)_source, _desc, _msg->data, _data); \
217 } while (0)
