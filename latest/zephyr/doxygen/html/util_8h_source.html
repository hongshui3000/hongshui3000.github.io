240 static inline int64_t arithmetic_shift_right(int64_t value, uint8_t shift)
241 {
242  int64_t sign_ext;
243 
244  if (shift == 0U) {
245  return value;
246  }
247 
248  /* extract sign bit */
249  sign_ext = (value >> 63) & 1;
250 
251  /* make all bits of sign_ext be the same as the value's sign bit */
252  sign_ext = -sign_ext;
253 
254  /* shift value and fill opened bit positions with sign bit */
255  return (value >> shift) | (sign_ext << (64 - shift));
256 }
