170 static inline int adc_raw_to_millivolts(int32_t ref_mv,
171  enum adc_gain gain,
172  uint8_t resolution,
173  int32_t *valp)
174 {
175  int32_t adc_mv = *valp * ref_mv;
176  int ret = adc_gain_invert(gain, &adc_mv);
177 
178  if (ret == 0) {
179  *valp = (adc_mv >> resolution);
180  }
181 
182  return ret;
183 }
