55 /* n is octal pin number or equivalent in another base.
56  * MCHP XEC documentation specifies pin numbers in octal.
57  * f is function number
58  * b[3:0] = pin bank
59  * b[8:4] = pin position in bank
60  * b[11:9] = function
61  */
62 #define MCHP_XEC_PINMUX(n, f) \
63  (((((n) >> 5) & MCHP_XEC_PINMUX_PORT_MSK) << MCHP_XEC_PINMUX_PORT_POS) | \
64  (((n) & MCHP_XEC_PINMUX_PIN_MSK) << MCHP_XEC_PINMUX_PIN_POS) | \
65  (((f) & MCHP_XEC_PINMUX_FUNC_MSK) << MCHP_XEC_PINMUX_FUNC_POS))
